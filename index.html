function loadFromLocalStorage() {

try {

const savedUpvotes = localStorage.getItem('sel-upvotes');

const savedComments = localStorage.getItem('sel-comments');

STATE.upvotes = savedUpvotes ? JSON.parse(savedUpvotes) : { ...INITIAL_UPVOTES };

STATE.comments = savedComments ? JSON.parse(savedComments) : JSON.parse(JSON.stringify(INITIAL_COMMENTS));

} catch (e) {

console.error('Error loading from localStorage:', e);

STATE.upvotes = { ...INITIAL_UPVOTES };

STATE.comments = JSON.parse(JSON.stringify(INITIAL_COMMENTS));

}

}

 

function saveToLocalStorage() {

try {

localStorage.setItem('sel-upvotes', JSON.stringify(STATE.upvotes));

localStorage.setItem('sel-comments', JSON.stringify(STATE.comments));

} catch (e) {

console.error('Error saving to localStorage:', e);

}

}

 

function escapeHtml(text) {

const div = document.createElement('div');

div.textContent = text;

return div.innerHTML;

}

 

function sanitizeId(name) {

return name.replace(/\s+/g, '-').replace(/[()]/g, '').toLowerCase();

}

 

// ============================================

// FILTER MANAGEMENT

// ============================================

function initializeFilters() {

const allBeneficiaries = [...new Set(TOOLKITS_DATA.flatMap(t => t.beneficiaries))].sort();

const allImplementers = [...new Set(TOOLKITS_DATA.flatMap(t => t.implementers))].sort();

const allEnvironments = [...new Set(TOOLKITS_DATA.flatMap(t => t.environment))].sort();

const allAdaptability = [...new Set(TOOLKITS_DATA.map(t => t.adaptability))].sort();

const allFocusAreas = [...new Set(TOOLKITS_DATA.flatMap(t => t.focusAreas))].sort();

 

createFilterButtons('beneficiariesFilters', allBeneficiaries, 'beneficiaries');

createFilterButtons('implementersFilters', allImplementers, 'implementers');

createFilterButtons('environmentFilters', allEnvironments, 'environment');

createFilterButtons('adaptabilityFilters', allAdaptability, 'adaptability');

createFilterButtons('focusAreasFilters', allFocusAreas, 'focusAreas');

}

 

function createFilterButtons(containerId, options, filterType) {

const container = document.getElementById(containerId);

if (!container) return;

container.innerHTML = '';

options.forEach(option => {

const btn = document.createElement('button');

btn.textContent = option;

btn.className = 'px-3 py-1 rounded-full text-xs transition-colors bg-gray-100 text-gray-700 hover:bg-gray-200 focus:outline-none focus:ring-2 focus:ring-blue-500';

btn.setAttribute('data-filter-type', filterType);

btn.setAttribute('data-filter-value', option);

btn.onclick = () => toggleFilter(filterType, option, btn);

container.appendChild(btn);

});

}

 

function toggleFilter(filterType, value, btn) {

const index = STATE.filters[filterType].indexOf(value);

if (index > -1) {

STATE.filters[filterType].splice(index, 1);

btn.className = 'px-3 py-1 rounded-full text-xs transition-colors bg-gray-100 text-gray-700 hover:bg-gray-200 focus:outline-none focus:ring-2 focus:ring-blue-500';

} else {

STATE.filters[filterType].push(value);

btn.className = 'px-3 py-1 rounded-full text-xs transition-colors bg-blue-600 text-white focus:outline-none focus:ring-2 focus:ring-blue-700';

}

renderToolkits();

updateFilterCount();

}

 

function clearAllFilters() {

STATE.filters = {

search: '',

beneficiaries: [],

implementers: [],

environment: [],

adaptability: [],

focusAreas: []

};

document.getElementById('searchInput').value = '';

document.querySelectorAll('#filtersPanel button').forEach(btn => {

btn.className = 'px-3 py-1 rounded-full text-xs transition-colors bg-gray-100 text-gray-700 hover:bg-gray-200 focus:outline-none focus:ring-2 focus:ring-blue-500';

});

renderToolkits();

updateFilterCount();

}

 

function updateFilterCount() {

const count = STATE.filters.beneficiaries.length +

STATE.filters.implementers.length +

STATE.filters.environment.length +

STATE.filters.adaptability.length +

STATE.filters.focusAreas.length;

const filterCountEl = document.getElementById('filterCount');

const clearBtn = document.getElementById('clearBtn');

if (count > 0) {

filterCountEl.textContent = count;

filterCountEl.classList.remove('hidden');

clearBtn.classList.remove('hidden');

} else {

filterCountEl.classList.add('hidden');

clearBtn.classList.add('hidden');

}

}

 

function filterToolkits() {

return TOOLKITS_DATA.filter(toolkit => {

const searchLower = STATE.filters.search.toLowerCase();

const matchesSearch = !searchLower ||

toolkit.name.toLowerCase().includes(searchLower) ||

toolkit.composition.toLowerCase().includes(searchLower) ||

toolkit.focusAreas.some(area => area.toLowerCase().includes(searchLower));

const matchesBeneficiaries = STATE.filters.beneficiaries.length === 0 ||

STATE.filters.beneficiaries.some(b => toolkit.beneficiaries.includes(b));

const matchesImplementers = STATE.filters.implementers.length === 0 ||

STATE.filters.implementers.some(i => toolkit.implementers.includes(i));

const matchesEnvironment = STATE.filters.environment.length === 0 ||

STATE.filters.environment.some(e => toolkit.environment.includes(e));

const matchesAdaptability = STATE.filters.adaptability.length === 0 ||

STATE.filters.adaptability.includes(toolkit.adaptability);

const matchesFocusAreas = STATE.filters.focusAreas.length === 0 ||

STATE.filters.focusAreas.some(f => toolkit.focusAreas.includes(f));

 

return matchesSearch && matchesBeneficiaries && matchesImplementers &&

matchesEnvironment && matchesAdaptability && matchesFocusAreas;

});

}

 

// ============================================

// INTERACTION HANDLERS

// ============================================

function handleUpvote(toolkitName) {

STATE.upvotes[toolkitName] = (STATE.upvotes[toolkitName] || 0) + 1;

const upvoteEl = document.getElementById(`upvote-${sanitizeId(toolkitName)}`);

if (upvoteEl) {

upvoteEl.textContent = STATE.upvotes[toolkitName];

upvoteEl.parentElement.classList.add('scale-110');

setTimeout(() => upvoteEl.parentElement.classList.remove('scale-110'), 200);

}

saveToLocalStorage();

}

 

function toggleComments(toolkitName) {

STATE.expandedComments[toolkitName] = !STATE.expandedComments[toolkitName];

const commentsSection = document.getElementById(`comments-${sanitizeId(toolkitName)}`);

if (commentsSection) {

commentsSection.classList.toggle('hidden');

}

}

 

function addComment(toolkitName) {

const inputId = `comment-input-${sanitizeId(toolkitName)}`;

const input = document.getElementById(inputId);

if (!input) return;

const text = input.value.trim();

if (!text) return;

if (!STATE.comments[toolkitName]) {

STATE.comments[toolkitName] = [];

}

STATE.comments[toolkitName].push({

text: text,

timestamp: new Date().toISOString(),

id: Date.now()

});

input.value = '';

saveToLocalStorage();

renderToolkits();

// Re-expand comments section

STATE.expandedComments[toolkitName] = true;

const commentsSection = document.getElementById(`comments-${sanitizeId(toolkitName)}`);

if (commentsSection) {

commentsSection.classList.remove('hidden');

}

}

 

// ============================================

// RENDERING FUNCTIONS

// ============================================

function renderToolkits() {

const filtered = filterToolkits();

const grid = document.getElementById('toolkitGrid');

const noResults = document.getElementById('noResults');

const resultCount = document.getElementById('resultCount');

if (!grid || !noResults || !resultCount) return;

resultCount.textContent = filtered.length;

if (filtered.length === 0) {

grid.classList.add('hidden');

noResults.classList.remove('hidden');

return;

}

grid.classList.remove('hidden');

noResults.classList.add('hidden');

grid.innerHTML = '';

filtered.forEach(toolkit => {

const card = createToolkitCard(toolkit);

grid.appendChild(card);

});

}

// ============================================

// EVENT LISTENERS

// ============================================

function setupEventListeners() {

// Search input

const searchInput = document.getElementById('searchInput');

if (searchInput) {

searchInput.addEventListener('input', (e) => {

STATE.filters.search = e.target.value;

renderToolkits();

});

}

 

// Filter button

const filterBtn = document.getElementById('filterBtn');

if (filterBtn) {

filterBtn.addEventListener('click', () => {

const panel = document.getElementById('filtersPanel');

if (panel) {

panel.classList.toggle('hidden');

}

});

}

 

// Clear button

const clearBtn = document.getElementById('clearBtn');

if (clearBtn) {

clearBtn.addEventListener('click', clearAllFilters);

}

}

 

// ============================================

// INITIALIZATION

// ============================================

function init() {

loadFromLocalStorage();

initializeFilters();

renderToolkits();

setupEventListeners();

}

 

// Initialize when DOM is ready

if (document.readyState === 'loading') {

document.addEventListener('DOMContentLoaded', init);

} else {

init();

}

 

// Handle page visibility change (refresh data when user returns to tab)

document.addEventListener('visibilitychange', () => {

if (!document.hidden) {

loadFromLocalStorage();

renderToolkits();

}

});
