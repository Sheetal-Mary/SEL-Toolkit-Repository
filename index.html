class SELDashboard {

constructor() {

this.state = {

searchTerm: '',

selectedBeneficiaries: [],

selectedImplementers: [],

selectedEnvironment: [],

selectedAdaptability: [],

selectedFocusAreas: [],

showFilters: false,

upvotes: { ...initialUpvotes },

comments: JSON.parse(JSON.stringify(initialComments)),

newComment: {},

expandedComments: {}

};

this.allBeneficiaries = [...new Set(toolkits.flatMap(t => t.beneficiaries))].sort();

this.allImplementers = [...new Set(toolkits.flatMap(t => t.implementers))].sort();

this.allEnvironments = [...new Set(toolkits.flatMap(t => t.environment))].sort();

this.allAdaptability = [...new Set(toolkits.map(t => t.adaptability))].sort();

this.allFocusAreas = [...new Set(toolkits.flatMap(t => t.focusAreas))].sort();

this.render();

}

 

getFilteredToolkits() {

return toolkits.filter(toolkit => {

const matchesSearch = toolkit.name.toLowerCase().includes(this.state.searchTerm.toLowerCase()) ||

toolkit.composition.toLowerCase().includes(this.state.searchTerm.toLowerCase()) ||

toolkit.focusAreas.some(area => area.toLowerCase().includes(this.state.searchTerm.toLowerCase()));

const matchesBeneficiaries = this.state.selectedBeneficiaries.length === 0 ||

this.state.selectedBeneficiaries.some(b => toolkit.beneficiaries.includes(b));

const matchesImplementers = this.state.selectedImplementers.length === 0 ||

this.state.selectedImplementers.some(i => toolkit.implementers.includes(i));

const matchesEnvironment = this.state.selectedEnvironment.length === 0 ||

this.state.selectedEnvironment.some(e => toolkit.environment.includes(e));

const matchesAdaptability = this.state.selectedAdaptability.length === 0 ||

this.state.selectedAdaptability.includes(toolkit.adaptability);

const matchesFocusAreas = this.state.selectedFocusAreas.length === 0 ||

this.state.selectedFocusAreas.some(f => toolkit.focusAreas.includes(f));

 

return matchesSearch && matchesBeneficiaries && matchesImplementers &&

matchesEnvironment && matchesAdaptability && matchesFocusAreas;

});

}

 

toggleFilter(value, filterType) {

const selected = this.state[filterType];

if (selected.includes(value)) {

this.state[filterType] = selected.filter(v => v !== value);

} else {

this.state[filterType] = [...selected, value];

}

this.render();

}

 

clearAllFilters() {

this.state.searchTerm = '';

this.state.selectedBeneficiaries = [];

this.state.selectedImplementers = [];

this.state.selectedEnvironment = [];

this.state.selectedAdaptability = [];

this.state.selectedFocusAreas = [];

this.render();

}

 

handleUpvote(toolkitName) {

this.state.upvotes[toolkitName] = (this.state.upvotes[toolkitName] || 0) + 1;

this.render();

}

 

handleAddComment(toolkitName) {

if (this.state.newComment[toolkitName]?.trim()) {

if (!this.state.comments[toolkitName]) {

this.state.comments[toolkitName] = [];

}

this.state.comments[toolkitName].push({

text: this.state.newComment[toolkitName],

timestamp: new Date().toISOString(),

id: Date.now()

});

this.state.newComment[toolkitName] = '';

this.render();

}

}

 

toggleComments(toolkitName) {

this.state.expandedComments[toolkitName] = !this.state.expandedComments[toolkitName];

this.render();

}

 

renderFilterSection(title, options, filterType) {

const selected = this.state[filterType];

return `

<div class="mb-4">

<h3 class="font-semibold text-sm text-gray-700 mb-2">${title}</h3>

<div class="flex flex-wrap gap-2">

${options.map(option => `

<button

onclick="dashboard.toggleFilter('${option}', '${filterType}')"

class="px-3 py-1 rounded-full text-xs transition-colors ${

selected.includes(option)

? 'bg-blue-600 text-white'

: 'bg-gray-100 text-gray-700 hover:bg-gray-200'

}"

>

${option}

</button>

`).join('')}

</div>

</div>

`;

}

this.state.selectedFocusAreas.length;

 

const html = `

Clear

>

</button>

</div>

</div>

<div class="space-y-3 text-sm">

${toolkit.beneficiaries.length > 0 ? `

<div>

<span class="font-semibold text-gray-700">Beneficiaries:</span>

<div class="flex flex-wrap gap-1 mt-1">

${toolkit.beneficiaries.map(b => `
